## Quick Example

```typescript
import { fetcher } from '@shkumbinhsn/fetcher';
import { z } from 'zod';

const UserSchema = z.object({
  id: z.string(),
  name: z.string(),
  email: z.string().email()
});

// Fully typed response
const user = await fetcher('/api/users/123', {
  schema: UserSchema
});

console.log(user.name); // TypeScript knows this is a string
```
